function parseChatMessage(input) {
  const regex = /^\[(\d{2}\/\d{2}\/\d{4}), (\d{2}:\d{2}:\d{2})\]\s+(.*?)\:\s+(.*)/;
  const match = input.match(regex);
  if (!match) return null;
  const [, date, time, sender, message] = match;
  return { date, time, sender, message };
}
